function toggleAnswer(btn) {
  const answer = btn.nextElementSibling;
  answer.classList.toggle('show');
  btn.textContent = answer.classList.contains('show') 
    ? '隐藏答案' 
    : '查看答案';
}
```

1. 添加到 quiz.html
2. 给按钮加 onclick="toggleAnswer(this)"
3. 答案区域加 class="answer-content"
